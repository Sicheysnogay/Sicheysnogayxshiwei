-- Font Tab
local FontTab = Window:MakeTab({
    Name = "Font",
    Icon = "rbxassetid://6031071053",
    PremiumOnly = false
})
local fonts = {
    {"Arial", Enum.Font.Arial},
    {"Georgia", Enum.Font.Fantasy},
    {"Courier New", Enum.Font.Code},
    {"Verdana", Enum.Font.Roboto},
    {"Helvetica", Enum.Font.Gotham},
    {"Minecraft", Enum.Font.Arcade}
}
for _, font in ipairs(fonts) do
    FontTab:AddButton({
        Name = font[1],
        Callback = function()
            for _, v in ipairs(game:GetDescendants()) do
                if v:IsA("TextLabel") or v:IsA("TextButton") or v:IsA("TextBox") then
                    v.Font = font[2]
                end
            end
        end
    })
end

-- Graphics Tab
local GraphicsTab = Window:MakeTab({
    Name = "Graphics",
    Icon = "rbxassetid://6031071053",
    PremiumOnly = false
})
GraphicsTab:AddSlider({
    Name = "Độ hoạ (20-90)",
    Min = 20,
    Max = 90,
    Default = 60,
    Callback = function(value)
        settings().Rendering.QualityLevel = Enum.QualityLevel.Level01 + math.floor(value/10)
    end
})
GraphicsTab:AddButton({
    Name = "Xoá sương mù",
    Callback = function()
        if game.Lighting:FindFirstChildOfClass("Atmosphere") then
            game.Lighting:FindFirstChildOfClass("Atmosphere"):Destroy()
        end
        game.Lighting.FogEnd = 1e9
    end
})
GraphicsTab:AddButton({
    Name = "Boost đồ hoạ",
    Callback = function()
        game.Lighting.Brightness = 3
        game.Lighting.GlobalShadows = true
        game.Lighting.Ambient = Color3.fromRGB(128,128,128)
    end
})

-- FPS Tab
local FPSTab = Window:MakeTab({
    Name = "FPS",
    Icon = "rbxassetid://6031071053",
    PremiumOnly = false
})
local fpsLabel
FPSTab:AddToggle({
    Name = "Hiện FPS",
    Default = false,
    Callback = function(state)
        if state then
            local RunService = game:GetService("RunService")
            local gui = Instance.new("TextLabel")
            gui.Size = UDim2.new(0,200,0,50)
            gui.Position = UDim2.new(0,10,0,10)
            gui.TextColor3 = Color3.new(1,1,1)
            gui.BackgroundTransparency = 1
            gui.TextSize = 20
            gui.Font = Enum.Font.Gotham
            gui.Text = "FPS: ..."
            gui.Parent = game.CoreGui
            fpsLabel = gui
            local last = tick()
            local frames = 0
            RunService.RenderStepped:Connect(function()
                frames += 1
                if tick() - last >= 1 then
                    local fps = math.floor(frames / (tick()-last))
                    fpsLabel.Text = "FPS: "..fps
                    frames = 0
                    last = tick()
                end
            end)
        elseif fpsLabel then
            fpsLabel:Destroy()
        end
    end
})
